```ts
export async function getPosts() {
  return fs.readdir(POST_DIR).then((fileList) => {
    return Promise.all(
      fileList.map((file) => {
        return fs
          .readFile(path.join(POST_DIR, file), "utf8")
          .then((contents) => {
            return serialize(contents) || null;
          })
          .then((contents) => {
            return {
              slug: file.replace(".mdx", ""),
              filePath: file.replace(".mdx", ""),
              source: contents,
            };
          });
      })
    );
  });
}
```
